    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useRef } = React;

        // --- ICONS & ASSETS ---
        const Icon = ({ name, className }) => <i className={`ph-fill ph-${name} ${className}`}></i>;

        // --- MAIN APP COMPONENT ---
        function App() {
            // --- STATE ---
            const [messages, setMessages] = useState([
                { id: 1, sender: 'bot', text: "Karibu! ðŸ‘‹ Welcome to Nairobi Fashion Hub. I'm your AI assistant. Looking for anything specific today?" }
            ]);
            const [inputText, setInputText] = useState("");
            const [isTyping, setIsTyping] = useState(false);
            const [activeNode, setActiveNode] = useState(null); // Which Make.com node is working?
            
            // Dashboard Mock Data
            const [revenue, setRevenue] = useState(14500);
            const [inventory, setInventory] = useState({
                "Red Dress": 4,
                "Blue Denim": 12,
                "Sneakers": 8
            });
            const [orderStatus, setOrderStatus] = useState("Idle");

            const bottomRef = useRef(null);

            // --- AUTO SCROLL ---
            useEffect(() => {
                bottomRef.current?.scrollIntoView({ behavior: "smooth" });
            }, [messages, isTyping]);

            // --- LOGIC SIMULATION ---
            const handleSend = async () => {
                if (!inputText.trim()) return;
                
                const userMsg = { id: Date.now(), sender: 'user', text: inputText };
                setMessages(prev => [...prev, userMsg]);
                setInputText("");
                setIsTyping(true);

                // Simulate Network Delay & Logic
                processLogic(userMsg.text);
            };

            const processLogic = async (text) => {
                const lowerText = text.toLowerCase();
                
                // STEP 1: SENTIMENT & INTENT (OpenAI)
                setActiveNode("openai");
                await wait(1000);
                
                if (lowerText.includes("angry") || lowerText.includes("stupid") || lowerText.includes("refund")) {
                    // Sentiment: Negative
                    setIsTyping(false);
                    setActiveNode("human");
                    addBotMessage("I apologize for the frustration. ðŸš¨ I have tagged a human manager to call you immediately.");
                    return;
                }

                // STEP 2: INVENTORY CHECK (Loyverse)
                if (lowerText.includes("dress") || lowerText.includes("red")) {
                    setActiveNode("loyverse");
                    await wait(1000);
                    
                    if (inventory["Red Dress"] > 0) {
                        setIsTyping(false);
                        addBotMessage(`Yes! We have ${inventory["Red Dress"]} Red Dresses in stock. The price is KES 2,500. Shall I add it to your cart? ðŸ›’`);
                    } else {
                        setIsTyping(false);
                        addBotMessage("Oh no! The Red Dress is out of stock. ðŸ˜” Would you like to see the Blue Denim instead?");
                    }
                    setActiveNode(null);
                    return;
                }

                // STEP 3: PAYMENT (Paystack)
                if (lowerText.includes("yes") || lowerText.includes("buy") || lowerText.includes("take")) {
                    setIsTyping(false);
                    setActiveNode("manychat");
                    addBotMessage("Great choice! ðŸ›ï¸ Please click below to pay securely via M-Pesa/Card.");
                    
                    // Show Payment Button
                    setTimeout(() => {
                        setMessages(prev => [...prev, { 
                            id: Date.now(), 
                            sender: 'system', 
                            type: 'payment_link',
                            text: "Pay KES 2,500" 
                        }]);
                    }, 500);
                    return;
                }

                // STEP 4: ADMIN
                if (lowerText.includes("admin") || lowerText.includes("owner")) {
                     // Security Check Mock
                    setActiveNode("security");
                    await wait(800);
                    setIsTyping(false);
                    addBotMessage("ðŸ” Admin Mode Verified. \n\nToday's Sales: KES " + revenue + "\nPending Orders: 2\nLow Stock: Red Dress");
                    setActiveNode(null);
                    return;
                }

                // DEFAULT
                setIsTyping(false);
                setActiveNode("openai");
                addBotMessage("I can help you check stock, buy items, or track an order. Try saying 'Do you have a Red Dress?'");
                setTimeout(() => setActiveNode(null), 1000);
            };

            const handlePaymentClick = async () => {
                // Simulate Paystack Loop
                setActiveNode("paystack");
                setOrderStatus("Processing Payment...");
                await wait(1500);
                
                setActiveNode("paystack_loop");
                setOrderStatus("Waiting for PIN...");
                await wait(2000);
                
                setOrderStatus("Payment Confirmed!");
                setRevenue(prev => prev + 2500);
                setInventory(prev => ({...prev, "Red Dress": prev["Red Dress"] - 1}));
                
                // Trigger Delivery
                setActiveNode("uber");
                await wait(1000);
                addBotMessage("âœ… Payment Received! KES 2,500 confirmed.");
                await wait(800);
                addBotMessage("ðŸšš I've requested a rider via Uber Direct. Your driver 'Juma' is on the way! (ETA: 25 mins)");
                
                // Final Data Sync
                setActiveNode("airtable");
                setTimeout(() => setActiveNode(null), 2000);
            };

            const addBotMessage = (text) => {
                setMessages(prev => [...prev, { id: Date.now(), sender: 'bot', text }]);
            };

            const wait = (ms) => new Promise(resolve => setTimeout(resolve, ms));

            // --- RENDER ---
            return (
                <div className="min-h-screen flex flex-col">
                    {/* Header */}
                    <header className="bg-white border-b border-slate-200 py-4 px-6 sticky top-0 z-50 flex justify-between items-center">
                        <div className="flex items-center gap-2">
                            <div className="bg-green-600 text-white p-2 rounded-lg">
                                <Icon name="robot" className="text-xl" />
                            </div>
                            <div>
                                <h1 className="font-bold text-slate-800 text-lg leading-tight">RetailAI Automation</h1>
                                <p className="text-xs text-slate-500">Interactive Prototype V1.0</p>
                            </div>
                        </div>
                        <div className="flex gap-3">
                            <button onClick={() => window.open('https://make.com', '_blank')} className="text-sm px-4 py-2 bg-slate-100 hover:bg-slate-200 text-slate-700 rounded-full font-medium transition">
                                View Backend (Make)
                            </button>
                            <button className="text-sm px-4 py-2 bg-green-600 hover:bg-green-700 text-white rounded-full font-medium transition shadow-lg shadow-green-200">
                                Contact Developer
                            </button>
                        </div>
                    </header>

                    {/* Main Content Grid */}
                    <main className="flex-1 max-w-7xl mx-auto w-full p-6 grid grid-cols-1 lg:grid-cols-3 gap-8">
                        
                        {/* COLUMN 1: SIMULATOR (PHONE) */}
                        <div className="lg:col-span-1 flex justify-center items-start pt-4">
                            <div className="w-full max-w-[380px] h-[700px] bg-slate-900 rounded-[3rem] border-8 border-slate-900 shadow-2xl overflow-hidden relative">
                                {/* Notch */}
                                <div className="absolute top-0 left-1/2 -translate-x-1/2 w-32 h-6 bg-slate-900 rounded-b-xl z-20"></div>
                                
                                {/* App Header */}
                                <div className="bg-[#075E54] text-white p-4 pt-10 flex items-center gap-3">
                                    <div className="w-8 h-8 bg-white/20 rounded-full flex items-center justify-center">
                                        <Icon name="storefront" className="text-lg" />
                                    </div>
                                    <div>
                                        <h3 className="font-semibold text-sm">Fashion Hub Bot</h3>
                                        <p className="text-[10px] text-white/80">Typing...</p>
                                    </div>
                                </div>

                                {/* Chat Area */}
                                <div className="bg-[#E5DDD5] h-full pb-32 overflow-y-auto phone-screen px-4 py-4 flex flex-col gap-3">
                                    {messages.map(msg => (
                                        <div key={msg.id} className={`flex flex-col ${msg.sender === 'user' ? 'items-end' : 'items-start'}`}>
                                            {msg.type === 'payment_link' ? (
                                                <button 
                                                    onClick={handlePaymentClick}
                                                    className="bg-white p-3 rounded-lg shadow-sm border-l-4 border-yellow-500 text-left w-48 hover:bg-slate-50 transition transform hover:scale-105"
                                                >
                                                    <div className="flex justify-between items-center mb-1">
                                                        <span className="font-bold text-slate-800">Invoice #001</span>
                                                        <Icon name="credit-card" className="text-yellow-600" />
                                                    </div>
                                                    <div className="text-sm text-slate-600">{msg.text}</div>
                                                    <div className="mt-2 text-xs bg-yellow-100 text-yellow-800 inline-block px-2 py-1 rounded font-medium">Tap to Pay</div>
                                                </button>
                                            ) : (
                                                <div className={`max-w-[85%] p-3 rounded-lg text-sm shadow-sm leading-relaxed ${
                                                    msg.sender === 'user' 
                                                    ? 'bg-[#DCF8C6] text-slate-800 rounded-tr-none' 
                                                    : 'bg-white text-slate-800 rounded-tl-none'
                                                }`}>
                                                    {msg.text}
                                                    <div className="text-[10px] text-right opacity-50 mt-1">{new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}</div>
                                                </div>
                                            )}
                                        </div>
                                    ))}
                                    
                                    {isTyping && (
                                        <div className="bg-white p-3 rounded-lg rounded-tl-none self-start w-16 flex items-center justify-center gap-1 shadow-sm">
                                            <div className="w-1.5 h-1.5 bg-slate-400 rounded-full typing-dot"></div>
                                            <div className="w-1.5 h-1.5 bg-slate-400 rounded-full typing-dot"></div>
                                            <div className="w-1.5 h-1.5 bg-slate-400 rounded-full typing-dot"></div>
                                        </div>
                                    )}
                                    <div ref={bottomRef}></div>
                                </div>

                                {/* Input Area */}
                                <div className="absolute bottom-0 w-full bg-[#f0f2f5] p-3 flex items-center gap-2">
                                    <input 
                                        type="text" 
                                        className="flex-1 bg-white rounded-full px-4 py-2 text-sm focus:outline-none border border-transparent focus:border-green-500"
                                        placeholder="Type a message..."
                                        value={inputText}
                                        onChange={(e) => setInputText(e.target.value)}
                                        onKeyPress={(e) => e.key === 'Enter' && handleSend()}
                                    />
                                    <button 
                                        onClick={handleSend}
                                        className="bg-[#075E54] text-white w-10 h-10 rounded-full flex items-center justify-center shadow-md hover:bg-[#054c44] transition"
                                    >
                                        <Icon name="paper-plane-right" className="text-lg" />
                                    </button>
                                </div>
                            </div>
                        </div>

                        {/* COLUMN 2: LOGIC FLOW (MAKE.COM) */}
                        <div className="lg:col-span-1 flex flex-col gap-4">
                            <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200 h-full">
                                <h2 className="text-slate-800 font-bold mb-6 flex items-center gap-2">
                                    <Icon name="circuitry" className="text-purple-600" />
                                    Logic Flow (Make.com)
                                </h2>
                                
                                <div className="space-y-4 relative">
                                    {/* Connecting Line */}
                                    <div className="absolute left-6 top-4 bottom-4 w-0.5 bg-slate-100 -z-0"></div>

                                    <LogicNode 
                                        id="manychat" 
                                        active={activeNode === "manychat"} 
                                        icon="whatsapp-logo" 
                                        color="text-green-500" 
                                        bg="bg-green-50"
                                        title="ManyChat Trigger"
                                        desc="Receives customer message via WhatsApp API"
                                    />

                                    <LogicNode 
                                        id="openai" 
                                        active={activeNode === "openai"} 
                                        icon="brain" 
                                        color="text-emerald-500" 
                                        bg="bg-emerald-50"
                                        title="GPT-4o Brain"
                                        desc="Analyzes Intent & Sentiment"
                                    />

                                    <LogicNode 
                                        id="loyverse" 
                                        active={activeNode === "loyverse"} 
                                        icon="storefront" 
                                        color="text-orange-500" 
                                        bg="bg-orange-50"
                                        title="Loyverse Inventory"
                                        desc="Checks real-time stock levels"
                                    />

                                    <LogicNode 
                                        id="paystack" 
                                        active={activeNode === "paystack" || activeNode === "paystack_loop"} 
                                        icon="credit-card" 
                                        color="text-blue-500" 
                                        bg="bg-blue-50"
                                        title="Paystack Payments"
                                        desc={activeNode === "paystack_loop" ? "â³ Looping: Waiting for PIN..." : "Initiates STK Push to Phone"}
                                    />

                                    <LogicNode 
                                        id="uber" 
                                        active={activeNode === "uber"} 
                                        icon="moped" 
                                        color="text-slate-800" 
                                        bg="bg-slate-100"
                                        title="Uber Direct API"
                                        desc="Requests nearest rider automatically"
                                    />

                                    <LogicNode 
                                        id="airtable" 
                                        active={activeNode === "airtable"} 
                                        icon="database" 
                                        color="text-yellow-600" 
                                        bg="bg-yellow-50"
                                        title="Airtable CRM"
                                        desc="Logs sale & updates customer record"
                                    />

                                </div>
                            </div>
                        </div>

                        {/* COLUMN 3: LIVE DASHBOARD */}
                        <div className="lg:col-span-1 flex flex-col gap-6">
                            
                            {/* Revenue Card */}
                            <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200 relative overflow-hidden">
                                <div className="absolute top-0 right-0 p-4 opacity-10">
                                    <Icon name="chart-line-up" className="text-6xl text-green-600" />
                                </div>
                                <h3 className="text-slate-500 text-sm font-medium mb-1">Today's Revenue</h3>
                                <div className="text-3xl font-bold text-slate-800 flex items-center gap-2">
                                    KES {revenue.toLocaleString()}
                                    {activeNode === "paystack" && <span className="text-xs bg-green-100 text-green-700 px-2 py-1 rounded animate-pulse">+ incoming</span>}
                                </div>
                                <p className="text-green-600 text-sm mt-2 flex items-center gap-1">
                                    <Icon name="trend-up" /> +12% from yesterday
                                </p>
                            </div>

                            {/* Inventory Card */}
                            <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200">
                                <h3 className="text-slate-800 font-bold mb-4 flex justify-between items-center">
                                    <span>Live Inventory (Loyverse)</span>
                                    <div className={`w-2 h-2 rounded-full ${activeNode === 'loyverse' ? 'bg-orange-500 animate-ping' : 'bg-green-500'}`}></div>
                                </h3>
                                <div className="space-y-4">
                                    {Object.entries(inventory).map(([item, count]) => (
                                        <div key={item}>
                                            <div className="flex justify-between text-sm mb-1">
                                                <span className="text-slate-600">{item}</span>
                                                <span className={`font-medium ${count < 5 ? 'text-red-500' : 'text-slate-800'}`}>{count} left</span>
                                            </div>
                                            <div className="w-full bg-slate-100 rounded-full h-2">
                                                <div 
                                                    className={`h-2 rounded-full transition-all duration-500 ${count < 5 ? 'bg-red-500' : 'bg-blue-600'}`} 
                                                    style={{ width: `${(count / 20) * 100}%` }}
                                                ></div>
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            </div>

                            {/* Recent Activity Console */}
                            <div className="bg-slate-900 p-6 rounded-2xl shadow-sm border border-slate-800 text-slate-300 font-mono text-xs h-full max-h-[300px] overflow-hidden flex flex-col">
                                <h3 className="text-white font-bold mb-3 border-b border-slate-700 pb-2">System Logs</h3>
                                <div className="flex-1 overflow-y-auto space-y-2">
                                    <div className="opacity-50">10:00:01 - System Initialized</div>
                                    <div className="opacity-50">10:05:22 - Loyverse Sync Complete</div>
                                    {activeNode === 'openai' && <div className="text-emerald-400">> Analyzing user intent...</div>}
                                    {activeNode === 'loyverse' && <div className="text-orange-400">> Querying Inventory API...</div>}
                                    {activeNode === 'paystack' && <div className="text-blue-400">> Initializing STK Push...</div>}
                                    {activeNode === 'paystack_loop' && <div className="text-blue-300">> Polling Payment Status (Try 1/3)...</div>}
                                    {activeNode === 'uber' && <div className="text-white">> Booking Uber Direct (Type: Moto)...</div>}
                                    {activeNode === 'airtable' && <div className="text-yellow-400">> Writing Record ID #99281...</div>}
                                </div>
                            </div>

                        </div>

                    </main>
                </div>
            );
        }

        // --- HELPER COMPONENTS ---
        function LogicNode({ id, active, icon, color, bg, title, desc }) {
            return (
                <div className={`relative z-10 flex items-start gap-3 p-3 rounded-xl border-2 transition-all duration-300 ${active ? 'bg-white border-green-500 shadow-lg scale-105' : 'bg-white border-transparent hover:border-slate-100'}`}>
                    <div className={`w-10 h-10 rounded-full flex items-center justify-center shrink-0 ${bg} ${color}`}>
                        <Icon name={icon} className="text-xl" />
                    </div>
                    <div>
                        <h4 className={`font-bold text-sm ${active ? 'text-slate-900' : 'text-slate-600'}`}>{title}</h4>
                        <p className="text-xs text-slate-500">{desc}</p>
                    </div>
                    {active && (
                        <div className="absolute right-3 top-3">
                            <span className="flex h-3 w-3">
                                <span className="animate-ping absolute inline-flex h-full w-full rounded-full bg-green-400 opacity-75"></span>
                                <span className="relative inline-flex rounded-full h-3 w-3 bg-green-500"></span>
                            </span>
                        </div>
                    )}
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
